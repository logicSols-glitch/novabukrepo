        // Custom message box function (instead of alert)
        function showMessage(text, type) {
            const box = document.getElementById('message-box');
            box.innerText = text;
            box.className = 'mt-4 p-4 rounded-md text-center';
            box.style.display = 'block';

            if (type === 'success') {
                box.classList.add('bg-green-100', 'text-green-700', 'border', 'border-green-300');
            } else if (type === 'error') {
                box.classList.add('bg-red-100', 'text-red-700', 'border', 'border-red-300');
            } else {
                box.classList.add('bg-blue-100', 'text-blue-700', 'border', 'border-blue-300');
            }

            // Hide the message after 5 seconds
            setTimeout(() => {
                box.style.display = 'none';
            }, 5000);
        }

        // Basic mobile menu toggle (only opens/closes, no routing needed for single-page)
        document.addEventListener('DOMContentLoaded', () => {
            const mobileMenuButton = document.querySelector('button.md\\:hidden');
            const navLinksContainer = document.querySelector('div.hidden.md\\:flex').parentNode; // Parent is the nav
            
            // Create a dedicated mobile menu container
            const mobileMenu = document.createElement('div');
            mobileMenu.className = 'md:hidden absolute top-full left-0 w-full bg-white shadow-lg p-4 transition-all duration-300 ease-in-out h-0 overflow-hidden';
            mobileMenu.id = 'mobile-menu';
            
            // Clone links into the mobile menu
            const desktopLinks = document.querySelector('div.hidden.md\\:flex').cloneNode(true);
            desktopLinks.className = 'flex flex-col space-y-3';
            
            // Attach event listener to close menu when a link is clicked
            desktopLinks.querySelectorAll('a').forEach(link => {
                link.classList.remove('space-x-8');
                link.classList.add('block', 'p-2', 'rounded-md', 'hover:bg-gray-50');
                link.addEventListener('click', () => {
                    toggleMobileMenu();
                });
            });

            mobileMenu.appendChild(desktopLinks);
            navLinksContainer.appendChild(mobileMenu);

            function toggleMobileMenu() {
                const isExpanded = mobileMenuButton.getAttribute('aria-expanded') === 'true';
                if (!isExpanded) {
                    mobileMenu.classList.remove('h-0');
                    mobileMenu.classList.add('h-auto', 'py-4'); /* py-4 to give padding when open */
                    mobileMenuButton.setAttribute('aria-expanded', 'true');
                } else {
                    mobileMenu.classList.remove('h-auto', 'py-4');
                    mobileMenu.classList.add('h-0');
                    mobileMenuButton.setAttribute('aria-expanded', 'false');
                }
            }

            mobileMenuButton.addEventListener('click', toggleMobileMenu);
        });

    </script>
